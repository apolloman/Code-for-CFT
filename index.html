OOP in JavaScript

Using Object Literal Notation:

let obj = {
prop1: 1,
prop2: "String",
prop3: function(){}, //Or ES6 methods prop3()
prop4: {}
}

Using new Object() syntax:

let obj = new Object();
obj.prop1 = 1;
obj.prop2 = "string";
obj.prop3 = function(){};
obj.prop4 = {};

Using Object Constructor

function Obj(firstName, lastName){
this.firstName = firstName;
this.lastName = lastName;
this.getAllName = function(){
return `${this.firstName} ${this.lastName};
}
}
let person1 = new Obj('Matthew', 'Fox');
let person2 = new Obj('Daniel', 'Craig');

Using Object.create()

let Human = {
power: 10,
intelligence: "different for all",
type: "man or woman",
displayType: function(){
console.log(this.type);
}
}
let human1 = Object.create(Human);
human1.displayType();

Inheritance in JavaScript

In JavaScript, inheritance is implemented with prototyping.

function Person(){}
Person.prototype.run = function(){};
function Animal(){}
Animal.prototype = new Person(); // Or Person.prototype
let example = new Animal();
Check:
console.log(example instanceof Animal);
console.log(example instanceof Person);
console.log(example instanceof Object);
